// Abbreviation generation logic:
function generateAbbreviation(name) {
  if (!name) return 'STN';
  const words = name.trim().split(/\s+/);
  if (words.length === 1) {
    return words[0].substring(0, 3).toUpperCase(); // "Wavelength" ‚Üí "WAV"
  } else {
    return words.slice(0, 4).map(word => word[0]).join('').toUpperCase(); // "Mumbai Midnight Beats" ‚Üí "MMB"
  }
}
